# Minimal example
from plantseg.predictions.functional.predictions import unet_predictions
import numpy as np

raw = np.random.random((128, 256, 256))
predictions = unet_predictions(raw,
                               model_name='generic_confocal_3d_unet',
                               patch=(80, 160, 160),
